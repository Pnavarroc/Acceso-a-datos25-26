24. Buscar posts con comentarios de un usuario específico y mostrar solo esos
comentarios

db.blog_posts.aggregate([
	{$match: {"comments.creador": "Ana García"}},
	{
	$project:{
		creador:1,
		contenido: 1,
		comments: {
			$filter: {
				input: "$comments",
				as: "comment",
				cond: {$eq: ["$$comment.creador", "Ana García"]}
			}
		}
	}
	}

])
